Here is a complete, self-contained interactive web page that explains and animates the pipeline of what happens when you navigate to a URL.

Save the code below as an `.html` file (for example, `browser-pipeline.html`) and open it in your browser to step through the annotated process.

```html
<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Browser Navigation Pipeline</title>
    <style>
        body {
            font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, Helvetica, Arial, sans-serif;
            background-color: #f4f4f9;
            color: #333;
            display: flex;
            flex-direction: column;
            align-items: center;
            padding: 40px 20px;
            margin: 0;
        }
        h1 {
            margin-bottom: 40px;
            text-align: center;
        }
        .pipeline {
            display: flex;
            align-items: center;
            justify-content: center;
            flex-wrap: wrap;
            gap: 15px;
            max-width: 900px;
            margin-bottom: 40px;
        }
        .stage {
            background: #fff;
            border: 2px solid #ccc;
            border-radius: 8px;
            padding: 15px 20px;
            font-weight: bold;
            color: #666;
            transition: all 0.3s ease;
            position: relative;
        }
        .stage.active {
            border-color: #007bff;
            background: #e6f2ff;
            color: #007bff;
            transform: scale(1.1);
            box-shadow: 0 4px 10px rgba(0, 123, 255, 0.2);
            z-index: 10;
        }
        .stage.completed {
            border-color: #28a745;
            background: #e6ffe6;
            color: #28a745;
        }
        .arrow {
            font-size: 24px;
            color: #ccc;
            transition: color 0.3s ease;
        }
        .arrow.active {
            color: #007bff;
        }
        .arrow.completed {
            color: #28a745;
        }
        .annotation-box {
            background: #fff;
            border-left: 5px solid #007bff;
            border-radius: 4px;
            padding: 20px;
            width: 100%;
            max-width: 600px;
            box-shadow: 0 2px 8px rgba(0,0,0,0.1);
            min-height: 150px;
            margin-bottom: 30px;
        }
        .annotation-title {
            font-size: 1.2em;
            font-weight: bold;
            margin-bottom: 10px;
            color: #007bff;
        }
        .annotation-text {
            line-height: 1.6;
        }
        .controls {
            display: flex;
            gap: 15px;
        }
        button {
            padding: 10px 20px;
            font-size: 16px;
            border: none;
            border-radius: 5px;
            cursor: pointer;
            background-color: #007bff;
            color: #fff;
            transition: background 0.2s;
        }
        button:hover {
            background-color: #0056b3;
        }
        button:disabled {
            background-color: #ccc;
            cursor: not-allowed;
        }
    </style>
</head>
<body>

    <h1>https://example.com Pipeline</h1>

    <div class="pipeline" id="pipeline">
        <!-- Generated by JS -->
    </div>

    <div class="annotation-box">
        <div class="annotation-title" id="anno-title">1. DNS Resolution</div>
        <div class="annotation-text" id="anno-text">Checking caches...</div>
    </div>

    <div class="controls">
        <button id="btn-prev" onclick="changeStage(-1)" disabled>Previous</button>
        <button id="btn-next" onclick="changeStage(1)">Next</button>
    </div>

    <script>
        const stages = [
            {
                name: "URL Parsing",
                title: "1. Parsing the URL & Checking HSTS",
                desc: "The browser extracts the protocol (https), domain (example.com), and path (/). It checks if the domain is in its Preloaded HSTS list to ensure it must use HTTPS. It also checks browser extensions and local cache for redirections."
            },
            {
                name: "DNS Resolution",
                title: "2. DNS Resolution",
                desc: "The browser needs the IP address of 'example.com'. It checks its own cache, the OS cache, the router cache, and finally asks the ISP's DNS resolver. The DNS queries route through the Root, TLD, and Authoritative name servers until it receives an IP address (e.g., 93.184.216.34)."
            },
            {
                name: "TCP Handshake",
                title: "3. TCP 3-Way Handshake",
                desc: "A reliable connection is established with the server at the IP address on port 443 (HTTPS). This uses a 3-way handshake:<br>1. <b>SYN</b>: Client asks to connect.<br>2. <b>SYN-ACK</b>: Server acknowledges and agrees.<br>3. <b>ACK</b>: Client acknowledges back. Connection opened!"
            },
            {
                name: "TLS Handshake",
                title: "4. TLS Handshake",
                desc: "Because we are using HTTPS, the connection must be encrypted.<br>1. Client sends 'ClientHello' (cipher suites).<br>2. Server replies with 'ServerHello' and its SSL certificate.<br>3. Browser verifies the certificate via Certificate Authorities.<br>4. Keys are exchanged to establish a symmetric encryption key for the session."
            },
            {
                name: "HTTP Request",
                title: "5. HTTP Request & Response",
                desc: "The browser constructs an HTTP GET request and sends it through the encrypted TLS tunnel. The server processes the request (routing, database lookups, etc.) and sends back an HTTP Response containing the status code (200 OK) and the payload (the HTML document)."
            },
            {
                name: "Rendering",
                title: "6. Browser Rendering",
                desc: "The browser parses the HTML to build the DOM tree. It requests linked assets (CSS, JS, Images). CSS builds the CSSOM tree. JS executes. The Render Tree is formed, layout is calculated (where elements go), and finally, the pixels are painted to the screen."
            }
        ];

        let currentStage = 0;

        function initPipeline() {
            const pipelineEl = document.getElementById('pipeline');
            stages.forEach((stage, index) => {
                const stageEl = document.createElement('div');
                stageEl.className = 'stage';
                stageEl.id = `stage-${index}`;
                stageEl.innerText = stage.name;
                pipelineEl.appendChild(stageEl);

                if (index < stages.length - 1) {
                    const arrowEl = document.createElement('div');
                    arrowEl.className = 'arrow';
                    arrowEl.id = `arrow-${index}`;
                    arrowEl.innerHTML = '&#10142;';
                    pipelineEl.appendChild(arrowEl);
                }
            });
            updateUI();
        }

        function changeStage(dir) {
            currentStage += dir;
            if (currentStage < 0) currentStage = 0;
            if (currentStage >= stages.length) currentStage = stages.length - 1;
            updateUI();
        }

        function updateUI() {
            // Update Text
            document.getElementById('anno-title').innerHTML = stages[currentStage].title;
            document.getElementById('anno-text').innerHTML = stages[currentStage].desc;

            // Update Pipeline Visuals
            stages.forEach((_, index) => {
                const stageEl = document.getElementById(`stage-${index}`);
                stageEl.classList.remove('active', 'completed');

                if (index < currentStage) stageEl.classList.add('completed');
                else if (index === currentStage) stageEl.classList.add('active');

                if (index < stages.length - 1) {
                    const arrowEl = document.getElementById(`arrow-${index}`);
                    arrowEl.classList.remove('active', 'completed');
                    if (index < currentStage) arrowEl.classList.add('completed');
                    else if (index === currentStage) arrowEl.classList.add('active');
                }
            });

            // Update Buttons
            document.getElementById('btn-prev').disabled = currentStage === 0;
            document.getElementById('btn-next').disabled = currentStage === stages.length - 1;
        }

        // Initialize on load
        window.onload = initPipeline;
    </script>
</body>
</html>
```
