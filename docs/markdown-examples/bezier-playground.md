Cubic Bézier curves are fundamental to computer graphics, animations, and UI design. They define a smooth curve using a mathematical formula and a set of four points:

1.  **$P_0$ (Start Point):** Where the curve begins.
2.  **$P_1$ (Control Point 1):** Determines the initial tangent or direction of the curve as it leaves $P_0$.
3.  **$P_2$ (Control Point 2):** Determines the incoming tangent or direction of the curve as it approaches $P_3$.
4.  **$P_3$ (End Point):** Where the curve ends.

In web development and animation (like CSS transitions), $P_0$ and $P_3$ are fixed at coordinates `(0,0)` and `(1,1)`. The curve's shape is controlled entirely by moving $P_1$ and $P_2$.

The curve is generated by a parametric function evaluating a parameter $t$ from $0$ (start) to $1$ (end). The math behind it uses Bernstein polynomials:

$$B(t) = (1-t)^3P_0 + 3(1-t)^2tP_1 + 3(1-t)t^2P_2 + t^3P_3$$

Here is a complete, self-contained interactive playground. Save the code below as an `.html` file and open it in your browser to experiment with dragging the control points and applying built-in presets.

            --curve-color: #ec4899;
        }
        body {
            font-family: system-ui, -apple-system, sans-serif;
            background-color: var(--bg-color);
            color: var(--text-color);
            display: flex;
            flex-direction: column;
            align-items: center;

        }

            box-shadow: 0 10px 25px rgba(0,0,0,0.05);

        }

<script>
    const canvas = document.getElementById('canvas');
    const h1 = document.getElementById('h1');
    const h2 = document.getElementById('h2');
    const curve = document.getElementById('curve');
    const line1 = document.getElementById('line1');
    const line2 = document.getElementById('line2');
    const output = document.getElementById('output');

    const size = 400;

    // Initial control points (standard ease)
    let p1 = { x: 0.25, y: 0.1 };
    let p2 = { x: 0.25, y: 1.0 };

    let dragging = null;

    function update() {
        // Map 0-1 coordinates to pixel dimensions (y is inverted in DOM)
        const x1 = p1.x * size;
        const y1 = (1 - p1.y) * size;
        const x2 = p2.x * size;
        const y2 = (1 - p2.y) * size;

        h1.style.left = `${x1}px`;
        h1.style.top  = `${y1}px`;
        h2.style.left = `${x2}px`;
        h2.style.top  = `${y2}px`;

        line1.setAttribute('d', `M 0 ${size} L ${x1} ${y1}`);
        line2.setAttribute('d', `M ${size} 0 L ${x2} ${y2}`);

        curve.setAttribute('d', `M 0 ${size} C ${x1} ${y1}, ${x2} ${y2}, ${size} 0`);

        output.textContent = `cubic-bezier(${p1.x.toFixed(2)}, ${p1.y.toFixed(2)}, ${p2.x.toFixed(2)}, ${p2.y.toFixed(2)})`;
    }

    function onMouseDown(e, handleStr) {
        dragging = handleStr;
        e.preventDefault();
    }

    function onMouseMove(e) {
        if (!dragging) return;

        const rect = canvas.getBoundingClientRect();
        let x = (e.clientX - rect.left) / size;
        let y = 1 - ((e.clientY - rect.top) / size);

        // Constrain X to 0-1 (CSS requirement for bezier timing functions)
        x = Math.max(0, Math.min(1, x));

        if (dragging === 'p1') {
            p1.x = x; p1.y = y;
        } else {
            p2.x = x; p2.y = y;
        }
        update();
    }

    function onMouseUp() {
        dragging = null;
    }

    h1.addEventListener('mousedown', (e) => onMouseDown(e, 'p1'));
    h2.addEventListener('mousedown', (e) => onMouseDown(e, 'p2'));
    document.addEventListener('mousemove', onMouseMove);
    document.addEventListener('mouseup', onMouseUp);

    window.setPreset = function(x1, y1, x2, y2) {
        p1 = { x: x1, y: y1 };
        p2 = { x: x2, y: y2 };
        update();
    }

    update();
</script>
</body>
</html>
